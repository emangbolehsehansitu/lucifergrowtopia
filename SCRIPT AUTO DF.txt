bot = getBot()
BotInventory = bot:getInventory()
world = bot:getWorld()

function clearLava()
   for _, tile in pairs(getTiles()) do
       while world:getTile(math.floor(getLocal().posx / 32),math.floor(getLocal().posy / 32) + 1).fg == 4 and BotInventory:getItemCount(2) > 0 do
           bot:findPath(tile.x,tile.y-1)
           punchbawah()
           sleep(160)
           placebawah(2,0,1)
       end
   end
end

function placebawah(itemID,xOffset, yOffset)
   local player = bot:getWorld():getLocal()
   local x = math.floor(player.posx / 32) + xOffset
   local y = math.floor(player.posy / 32) + yOffset
   bot:place(x, y, itemID)
end

function plantdirt()
   for i, tile in ipairs(world:getTiles()) do
      if 0 == world:getTile(tile.x, tile.y).fg and 0 ~= world:getTile(tile.x, tile.y + 1).fg and world:getTile(tile.x, tile.y + 1).fg ~= SeedID and BotInventory:getItemCount(SeedID) > 0 then
         bot:findPath(tile.x,tile.y)
         sleep(PlantDelay)
         if bot:isInTile(tile.x,tile.y) then
            bot:place(tile.x,tile.y,SeedID)
            sleep(PlantDelay)
            else
               bot:findPath(tile.x,tile.y)
               sleep(PlantDelay)
         if bot:isInTile(tile.x,tile.y) then
              bot:place(tile.x,tile.y,SeedID)
              sleep(PlantDelay)
           end
        end
      end
   end
end
function harvest()
   bot.auto_collect = true
   for i, tile in ipairs(world:getTiles()) do
      if tile.fg == SeedID and tile:canHarvest() then 
         bot:findPath(tile.x,tile.y)
         bot:hit(tile.x,tile.y)
         sleep(delayHarvest)
         if BotInventory:getItemCount(2) == 200 then
            break
         end
      end
   end
end

function punchbawah()
   for hit = 1,Block_Hit do
      bot:hit(math.floor(getLocal().posx / 32),math.floor(getLocal().posy / 32) + 1)
      sleep(Delay_Break)
   end
end

function punchkiri()
   for hit = 1,Block_Hit do
      bot:hit(math.floor(getLocal().posx / 32) - 1,math.floor(getLocal().posy / 32))
      sleep(Delay_Break)
   end
end

function punchkanan()
   for hit = 1,Block_Hit do
      bot:hit(math.floor(getLocal().posx / 32) + 1,math.floor(getLocal().posy / 32))
      sleep(Delay_Break)
   end
end

function CekClearSideKanan()
   for y = 23, 53 do
      local tile = getBot():getWorld():getTile(99, y)

      while tile.bg ~= 0 or tile.fg ~= 0 do
         bot:findPath(tile.x, tile.y - 1)
         punchbawah()
         tile = getBot():getWorld():getTile(99, y)
      end

      -- Setelah selesai memeriksa tile di sebelah kanan, periksa apakah ada yang belum bersih
      if tile.bg ~= 0 or tile.fg ~= 0 then
         return false -- Ada setidaknya satu tile yang belum bersih
      end
   end

   return true -- Semua tile di sebelah kanan sudah bersih
end


-- Fungsi untuk membersihkan sisi kiri
function CekClearSideKiri()
   bot.auto_collect = true
   for y = 23, 53 do
      tile = getBot():getWorld():getTile(0, y)
      while tile.bg ~= 0 or tile.fg ~= 0 do
         bot:findPath(tile.x, tile.y - 1)
         punchbawah()
      end
   end

   -- Setelah selesai periksa apakah sebelah kiri sudah bersih
   for y = 23, 53 do
      local tileKiri = getBot():getWorld():getTile(0, y)
      while tileKiri.bg ~= 0 or tileKiri.fg ~= 0 do
         return false  -- Ada tile yang belum bersih
      end
   end

   return true  -- Semua tile di sebelah kiri sudah bersih
end
